{{- /* Documentation version selector for the Hextra navbar. Data-driven from data/versions.toml. Renders a small dropdown that lists the moving CHANNELS first (latest / pre-release / stable) and then any frozen ARCHIVED concrete versions below a divider. Each entry links to its Pages subpath (e.g. /go-sdk/latest/, /go-sdk/stable/, /go-sdk/v0.0.1-alpha.5/). The currently-served version is detected from the request path so the trigger label reflects where the reader is. Hextra has no native doc versioning; this is the custom mechanism for AAASM-2754. The dropdown is pure CSS (hover + focus-within) so it needs no additional theme JavaScript. */ -}} {{- $versions := .Site.Data.versions.versions -}} {{- /* Version subpaths are absolute from the Pages site root. Prepend the configured Pages root path (params.versionsBasePath, e.g. "/go-sdk") so links resolve across versions. */ -}} {{- $base := .Site.Params.versionsBasePath | default "" -}} {{- $base = strings.TrimSuffix "/" $base -}} {{- if $versions -}} {{- /* Split entries into moving channels (rendered first) and archived. */ -}} {{- $channels := where $versions "channel" "!=" "archived" -}} {{- $archived := where $versions "channel" "archived" -}} {{- /* Determine the active version from the current URL path. Each per-version build serves under // (channels use their channel name as the subpath, archived use the bare tag), so match the version token in the path. */ -}} {{- $relPath := .RelPermalink -}} {{- $current := "" -}} {{- $currentLabel := "" -}} {{- range $versions -}} {{- if in $relPath (printf "/%s/" .version) -}} {{- $current = .version -}} {{- $currentLabel = .label -}} {{- end -}} {{- end -}} {{- if not $current -}} {{- range $versions -}} {{- if .default -}}{{- $current = .version -}}{{- $currentLabel = .label -}}{{- end -}} {{- end -}} {{- end -}} {{- if not $current -}} {{- $first := index $versions 0 -}} {{- $current = $first.version -}} {{- $currentLabel = $first.label -}} {{- end -}}

  • {{ .label }}
    • {{- end -}} {{- if $archived -}}
  • Archived
    • {{- range $archived -}}
  • {{ .label }}
